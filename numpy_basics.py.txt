import numpy as np


def array_statistics(pixel_values):
    """
    Given a list of pixel values, convert to a NumPy array and
    return a dictionary with basic statistics.

    Args:
        pixel_values (list): List of numeric pixel values

    Returns:
        dict: Dictionary with keys "min", "max", "mean", "std"
              All values should be rounded to 2 decimal places.

    Example:
        array_statistics([100, 150, 200, 50, 175])
        -> {"min": 50.0, "max": 200.0, "mean": 135.0, "std": 53.85}
    """
    # def array_statistics(pixel_values):
    arr = np.array(pixel_values)
    return {
        "min": round(np.min(arr), 2),
        "max": round(np.max(arr), 2),
        "mean": round(np.mean(arr), 2),
        "std": round(np.std(arr), 2)
    }
    pass


def normalize_pixels(pixel_values):
    """
    Normalize pixel values to range [0, 1] by dividing by 255.
    Return the result as a NumPy array rounded to 3 decimal places.

    Args:
        pixel_values (list): List of integer pixel values (0-255)

    Returns:
        numpy.ndarray: Array of normalized values rounded to 3 decimals

    Example:
        normalize_pixels([0, 128, 255]) -> array([0.0, 0.502, 1.0])
    """
    # def normalize_pixels(pixel_values):
    arr = np.array(pixel_values, dtype=float)
    return np.round(arr / 255, 3)
    pass


def extract_roi(image_2d, row_start, row_end, col_start, col_end):
    """
    Extract a Region of Interest (ROI) from a 2D NumPy array using slicing.

    Args:
        image_2d (numpy.ndarray): 2D array representing an image
        row_start (int): Starting row index (inclusive)
        row_end (int): Ending row index (exclusive)
        col_start (int): Starting column index (inclusive)
        col_end (int): Ending column index (exclusive)

    Returns:
        numpy.ndarray: The extracted ROI sub-array

    Example:
        image = np.array([[0, 0, 0],
                          [0, 50, 0],
                          [0, 0, 0]])
        extract_roi(image, 0, 2, 0, 2) -> array([[0, 0], [0, 50]])
    """
    # def extract_roi(image_2d, row_start, row_end, col_start, col_end):
    arr = np.array(image_2d)
    return arr[row_start:row_end, col_start:col_end]
    pass


def calculate_snr(signal_region, background_region):
    """
    Calculate the Signal-to-Noise Ratio (SNR).

    SNR = mean(signal_region) / std(background_region)

    Args:
        signal_region (list or numpy.ndarray): Pixel values from signal region
        background_region (list or numpy.ndarray): Pixel values from background

    Returns:
        float: SNR value rounded to 2 decimal places

    Example:
        signal = [180, 195, 200, 185, 190]
        background = [12, 8, 15, 10, 11]
        calculate_snr(signal, background) -> 82.07
    """
    # def calculate_snr(signal_region, background_region):
    signal = np.array(signal_region)
    background = np.array(background_region)
    return round(np.mean(signal) / np.std(background), 2)
    pass


def calculate_cnr(tissue_a, tissue_b, background):
    """
    Calculate the Contrast-to-Noise Ratio (CNR).

    CNR = |mean(tissue_a) - mean(tissue_b)| / std(background)

    Args:
        tissue_a (list or numpy.ndarray): Pixel values from tissue A (e.g., tumor)
        tissue_b (list or numpy.ndarray): Pixel values from tissue B (e.g., healthy)
        background (list or numpy.ndarray): Pixel values from background noise

    Returns:
        float: CNR value rounded to 2 decimal places

    Example:
        tumor = [200, 210, 195, 205]
        healthy = [120, 115, 125, 118]
        background = [12, 8, 15, 10]
        calculate_cnr(tumor, healthy, background) -> 32.1
    """
    # def calculate_cnr(tissue_a, tissue_b, background):
    tissue_a = np.array(tissue_a)
    tissue_b = np.array(tissue_b)
    background = np.array(background)
    return round(abs(np.mean(tissue_a) - np.mean(tissue_b)) / np.std(background), 2)
    pass
